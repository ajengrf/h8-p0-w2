2. Tahun Kabisat
SET tahun = 2019
CALCULATE tahun mod 4
SET hasil1 WITH RESULT OF CALCULATE tahun mod 4
IF hasil1 equals to 0
    CALCULATE tahun mod 100
    SET hasil2 WITH RESULT OF CALCULATE tahun mod 100
    IF hasil2 equals to 0
        CALCULATE tahun mod 400
        SET hasil3 WITH RESULT OF CALCULATE tahun mod 400
        IF hasil3 equals to 0
        DISPLAY Tahun Kabisat
        ELSE
         DISPLAY Bukan Tahun Kabisat
    ELSE
     DISPLAY Tahun Kabisat
ELSE
 DISPLAY Bukan Tahun Kabisat

ENDIF
